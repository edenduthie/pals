\HeaderA{color.id}{Identify closest match to a color}{color.id}
\keyword{color}{color.id}
\begin{Description}\relax
Given a color given as a hex string, find the closest match in the
table of known (named) colors
\end{Description}
\begin{Usage}
\begin{verbatim}
 color.id(col)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{col}] a color specified as a hex string
\end{ldescription}
\end{Arguments}
\begin{Details}\relax
finds the color with the minimum squared distance in RGB space
\end{Details}
\begin{Value}
the name of the closest match
\end{Value}
\begin{Author}\relax
Ben Bolker
\end{Author}
\begin{SeeAlso}\relax
\samp{\LinkA{col2rgb}{col2rgb}},\samp{\LinkA{colors}{colors}}
\end{SeeAlso}
\begin{Examples}
\begin{ExampleCode}
 color.id("#cc00cc")
\end{ExampleCode}
\end{Examples}
