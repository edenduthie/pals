\HeaderA{piper.set}{Calculate ion offsets.}{piper.set}
\keyword{misc}{piper.set}
\begin{Description}\relax
Calculates the offsets for anions and cations in the Piper
diagram.
\end{Description}
\begin{Usage}
\begin{verbatim}
 piper.set(plot.sep=0.15)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{plot.sep}] The offsets of the ions in user units.
\end{ldescription}
\end{Arguments}
\begin{Details}\relax
\samp{piper.set} calculates the ion offsets in the Piper diagram.
\end{Details}
\begin{Value}
The offsets.
\end{Value}
